\section{研究背景与动机}
% =====================================================

\begin{frame}{TCP/IP 跨层歧义持续催生 off-path 协议攻击}
    \begin{itemize}
        \item TCP/IP 协议栈的跨层交互存在\textbf{语义歧义}，长期催生 \alert{off-path}（链外）协议攻击——攻击者无需位于通信路径上，即可推断或注入连接状态：
        \begin{itemize}
            \item challenge-ACK \textbf{侧信道}\supercite{cao2016offpath}（CVE-2016-5696，全局限速可被探测）
            \item 混合 \textbf{IPID} 分配\supercite{feng2020mixedipid}、PMTUD 分片机制\supercite{feng2022pmtud}、NAT / Wi-Fi 链路劫持\supercite{yang2024natwifi}\,\ldots
        \end{itemize}
        \item 近期 \textbf{CCS'25} 工作\supercite{feng2025pmtudisolation}：利用 \textbf{PMTUD} 打破“IP 地址共享”场景下的 TCP 连接隔离
        \begin{itemize}
            \item 单次攻击平均约 \textbf{220\,s}，成功率约 \textbf{70\%}
            \item 实测 \textbf{50} 个真实网络中 \textbf{38} 个存在脆弱性
        \end{itemize}
    \end{itemize}
    \begin{block}{核心判断}
        协议攻击面并非偶发，而是植根于\textbf{设计层面的跨层歧义}，会被持续不断地重新发掘\supercite{feng2025cacm}。
    \end{block}
\end{frame}

\begin{frame}{规律：防御机制本身在制造新的攻击面}
    \begin{itemize}
        \item 大量新攻击面\textbf{恰由防御机制自身引入}——“补丁”反而成为下一处侧信道：
    \end{itemize}
    \vspace{0.3em}
    \begin{center}
    \begin{tikzpicture}[node distance=5mm and 9mm]
        \node[stepbox, fill=thupurple!4] (d1) {challenge-ACK\\ 全局限速};
        \node[stepbox, fill=thupurple!4, right=of d1] (d2) {源端口\\ 随机化};
        \node[stepbox, fill=thupurple!4, right=of d2] (d3) {PMTUD\\ 分片探测};
        \node[centerbox, below=9mm of d2] (atk) {均被转化为可探测的\,\textbf{侧信道 / 新攻击面}};
        \draw[flowarrow] (d1) -- (atk);
        \draw[flowarrow] (d2) -- (atk);
        \draw[flowarrow] (d3) -- (atk);
    \end{tikzpicture}
    \end{center}
    \vspace{0.2em}
    \begin{alertblock}{启示}
        攻击与防御\textbf{同源}且相互转化 $\Rightarrow$ 防御须\textbf{持续演化}，一次性静态修补难以为继。
    \end{alertblock}
\end{frame}

\begin{frame}{现有防御的三大结构性缺陷}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{block}{\cn{1} 零散}
                各攻击面分别打补丁，缺乏统一框架，防御策略碎片化、难以协同。
            \end{block}
            \begin{block}{\cn{2} 滞后}
                先有攻击曝光、再被动响应，从披露到修复存在显著时间窗口。
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{\cn{3} 依赖人工编写内核代码}
                \begin{itemize}
                    \item 内核协议栈改动\textbf{周期长}：评审、回归、上线缓慢
                    \item \textbf{易出错}：一处缺陷即可断网或误杀正常流量
                    \item 知识\textbf{难以沉淀}：经验留存于个人，复用困难
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
    \vspace{0.4em}
    \begin{center}
        \kaishu\color{thupurple}\textbf{症结：防御的生产方式仍是人工、低速、难以累积的。}
    \end{center}
\end{frame}

\begin{frame}{两项技术趋于成熟，使自动化防御成为可能}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{eBPF 内核可编程}
                \begin{itemize}
                    \item \textbf{热加载}：无需修改内核源码、无需重启
                    \item \textbf{贴近协议栈}：XDP / TC 钩子在数据面线速处理
                    \item 内置 \textbf{verifier}：对程序施加\textbf{形式化安全验证}（有界、内存安全、可终止）\supercite{gershuni2019prevail}
                \end{itemize}
            \end{exampleblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{大模型编码智能体}
                \begin{itemize}
                    \item 能够\textbf{编写、修改、调试}系统级代码
                    \item 具备工具调用、反思与迭代能力
                    \item 已能在反馈闭环中自主推进复杂任务\supercite{romeraparedes2024funsearch,novikov2025alphaevolve}
                    \item 代表性产品 / 框架：Claude Code、Cursor、Codex、Devin、SWE-agent\supercite{anthropic2025claudecode,yang2024sweagent}
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}
    \vspace{0.5em}
    \begin{block}{契机}
        二者结合:eBPF 作可热插拔、可验证的内核执行底座,智能体作自动化代码生产——使“由智能体持续维护内核防御”可行。
    \end{block}
\end{frame}

\section{研究问题与挑战}
% =====================================================

\begin{frame}{核心研究问题}
    \begin{block}{研究问题}
        \centering\large
        能否让\textbf{大模型编码智能体}自主维护一套运行于\textbf{内核}的 \textbf{eBPF 防御库}，\\[2pt]
        使其\textbf{持续进化}，以应对不断涌现的协议攻击？
    \end{block}
    \vspace{0.5em}
    \begin{itemize}
        \item 目标不是“一次性合成单个防御程序”，而是构建一个\textbf{长期自主运转}的防御系统；
        \item 核心在于\textbf{自主性}（探索—合成—验证—部署的闭环）与\textbf{进化性}（防御能力单调累积、不回退）。
    \end{itemize}
\end{frame}

\begin{frame}{四项关键挑战}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{\cn{1} 可靠性}
                智能体编写的内核代码若含缺陷，可能直接\textbf{断网}或\textbf{误杀}正常流量——必须确保“先验证安全、后部署”。
            \end{alertblock}
            \begin{alertblock}{\cn{2} 探索高效性}
                协议状态空间\textbf{巨大且充满噪声}，如何高效发现\textbf{真正有价值}的攻击面，而非无效扰动？
            \end{alertblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{\cn{3} 非梯度演化的收敛性}
                防御以\textbf{代码}为载体演化，如何保证其\textbf{单调进化}、不腐化、不回退（避免灾难性遗忘）？
            \end{alertblock}
            \begin{alertblock}{\cn{4} 评测}
                如何\textbf{量化}系统对\textbf{未知攻击}的自主防御能力，给出可复现、可对比的度量？
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\section{相关工作}
% =====================================================

\begin{frame}{相关工作（一）：协议攻防与 eBPF 运行时防御}
    \begin{block}{协议攻击与检测}
        \begin{itemize}
            \item \textbf{攻击}：off-path 协议攻击系列工作（challenge-ACK、IPID、PMTUD 等）\supercite{feng2025cacm,feng2025pmtudisolation}
            \item \textbf{检测}：SCENT\supercite{cao2019scent}、SCAD\supercite{man2024scad} 等侧信道检测方法
            \item \textbf{现状}：防御\textbf{零散、滞后}，缺少统一且持续演化的框架
        \end{itemize}
    \end{block}
    \begin{block}{eBPF 运行时防御}
        \begin{itemize}
            \item 系统：Katran\supercite{shirokov2018katran}、Cilium\supercite{graf2020cilium}、Falco\supercite{falco2018}、Tetragon\supercite{tetragon2022}——能力强，但策略\textbf{均由人工编写}
            \item 验证：PREVAIL\supercite{gershuni2019prevail}、K2\supercite{xu2021k2} 等围绕 verifier 的程序分析与优化
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{相关工作（二）：大模型用于安全 与 方法论基础}
    \begin{columns}[T]
        \begin{column}{0.52\textwidth}
            \begin{block}{大模型 $\times$ 安全}
                \begin{itemize}
                    \item \textbf{进攻}：PentestGPT\supercite{deng2024pentestgpt}、Big Sleep\supercite{glazunov2024naptime}
                    \item \textbf{合成 eBPF}：\alert{Kgent / KEN}\supercite{yang2024kgent}——由自然语言合成 eBPF，是\textbf{最直接的对比基线}
                    \begin{itemize}
                        \item 局限：\textbf{一次性合成}，\textbf{非自主、持续维护}
                    \end{itemize}
                    \item \textbf{工程载体}：Claude Code / Cursor / Codex / Devin / SWE-agent\supercite{anthropic2025claudecode,cursor2024,yang2024sweagent}——\textbf{通用而非领域专用}，需专门 harness
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{方法论基础}
                \begin{itemize}
                    \item 非梯度演化范式\supercite{weng2026hl}
                    \item 程序演化搜索：FunSearch\supercite{romeraparedes2024funsearch}、AlphaEvolve\supercite{novikov2025alphaevolve}
                    \item 自演化智能体：Darwin-G\"odel Machine\supercite{zhang2025dgm}
                    \item 探索机制：ICM\supercite{pathak2017icm}、RND\supercite{burda2019rnd}（基于预测误差与新颖性）
                    \item 多样化解集：Novelty Search\supercite{lehman2011novelty}、MAP-Elites\supercite{mouret2015mapelites}
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
    \vspace{0.2em}
    \begin{center}
        \footnotesize\color{thugray}上述方法论共同支撑“以可验证代码为载体的非梯度演化”。
    \end{center}
\end{frame}

\section{研究内容与方法}
% =====================================================

\begin{frame}{核心思路：防御演化（Defensive Evolution）}
    \begin{block}{什么是“防御演化”}
        让编码智能体\textbf{持续维护并演化}一套运行在内核的防御代码:把防御能力沉淀为\textbf{可验证的 eBPF 代码}(而不是模型权重),随攻击的出现不断增补、修正。
    \end{block}
    \vspace{0.2em}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{以可验证代码为载体的优势：}
            \begin{itemize}
                \item 容量\textbf{无界}（不受参数量限制）
                \item 抗\textbf{灾难性遗忘}（旧防御不被覆盖）
                \item \textbf{可解释}（每条防御对应一段代码）
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \vspace{1.6em}
            \begin{itemize}
                \item \textbf{线速}执行（XDP 数据面）
                \item 可\textbf{形式化验证}（verifier 背书）
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{四项研究内容}
    \begin{enumerate}\itemsep=0.5em
        \item \textbf{范式形式化 + 系统框架 + 专用 harness}\\
        \footnotesize\color{thugray}界定“以可验证代码为载体的演化”;把 eBPF 工具、verifier 反馈、沙箱、靶场、档案封装成智能体的工具接口。\normalsize\par
        \item \textbf{攻击面探索}\\
        \footnotesize\color{thugray}以新颖性与覆盖度为信号,发现现有防御覆盖不到的攻击。\normalsize\par
        \item \textbf{进化式 eBPF 合成 + 可信加载闭环}\\
        \footnotesize\color{thugray}大模型生成 / 改写 eBPF,经多层验证后热加载到内核。\normalsize\par
        \item \textbf{防御库维护 + 攻防协同演化评测}\\
        \footnotesize\color{thugray}沉淀多样化防御,量化对已知 / 未知攻击的防御效果。\normalsize\par
    \end{enumerate}
\end{frame}

\begin{frame}{评测设计：攻防协同演化}
    \begin{block}{评测方式}
        红队与防御系统\textbf{协同演化};攻击集分为\textbf{训练种子}与\textbf{未知攻击留出集},考察无人工补丁介入下的自主防御。
    \end{block}
    \begin{columns}[T]
        \begin{column}{0.52\textwidth}
            \textbf{度量指标}
            \begin{itemize}
                \item 已知攻击:\textbf{检出率 / 误报率}
                \item 未知攻击(留出):随迭代的\textbf{防御覆盖提升}
                \item 性能:XDP \textbf{吞吐损失 / 时延}
                \item \textbf{自动化程度}(人工介入量)
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{对比与消融}
            \begin{itemize}
                \item Kgent / KEN(一次性合成)
                \item Tetragon / Falco(人工策略)
                \item 消融:去探索 / 去档案 / 去自修正
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{创新点}
    \begin{exampleblock}{\cn{1} 范式}
        让编码智能体\textbf{持续维护与演化}内核防御代码——防御沉淀为“\textbf{可验证代码}”而非模型权重(即\textbf{防御演化})。

    \begin{exampleblock}{\cn{2} 专用 harness}
        为内核 eBPF 防御构建\textbf{领域专用的智能体 harness}(领域工具 + 反馈闭环),而非直接套用 Claude Code / Cursor 等通用编码智能体。
    \end{exampleblock}
    \begin{exampleblock}{\cn{3} 攻防协同演化评测基准}
        构建含\textbf{未知攻击留出集}的评测,量化对已知 / 未知攻击的\textbf{防御效果、误报与开销}。
    \end{exampleblock}
\end{frame}
